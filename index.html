Google Sheets — Conexión

Para sincronizar con Google Sheets necesitas un Apps Script Web App desplegado como API. Ver instrucciones ↗

Instrucciones: Google Apps Script
  1. Abre tu Google Sheet y ve a Extensiones → Apps Script
  2. Borra el contenido y pega el siguiente código:
// ═══════════════════════════════════════════
// APPS SCRIPT — Panel de Gestión
// Pega este código completo y despliega como Web App
// ═══════════════════════════════════════════

// ── IMPORTANTE: Este código maneja CORS para permitir
// que la app en Netlify pueda leer Y escribir en Sheets ──

function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss   = SpreadsheetApp.getActiveSpreadsheet();

    if (data.action === 'saveAll') {
      var shP = ss.getSheetByName('Pedidos') || ss.insertSheet('Pedidos');
      shP.clearContents();
      shP.getRange(1,1).setValue('datos_json');
      shP.getRange(2,1).setValue(JSON.stringify(data.pedidos || []));

      var shI = ss.getSheetByName('Inventario') || ss.insertSheet('Inventario');
      shI.clearContents();
      shI.getRange(1,1).setValue('datos_json');
      shI.getRange(2,1).setValue(JSON.stringify(data.inventario || []));
    }

    return ContentService
      .createTextOutput(JSON.stringify({ ok: true }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  try {
    var ss     = SpreadsheetApp.getActiveSpreadsheet();
    var result = { pedidos: [], inventario: [] };

    var shP = ss.getSheetByName('Pedidos');
    if (shP && shP.getLastRow() >= 2) {
      var rawP = shP.getRange(2,1).getValue();
      if (rawP) result.pedidos = JSON.parse(rawP);
    }

    var shI = ss.getSheetByName('Inventario');
    if (shI && shI.getLastRow() >= 2) {
      var rawI = shI.getRange(2,1).getValue();
      if (rawI) result.inventario = JSON.parse(rawI);
    }

    return ContentService
      .createTextOutput(JSON.stringify(result))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ pedidos: [], inventario: [], error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
